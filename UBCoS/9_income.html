The validation consists of checking just the most frequent trajectories of income.

# For 1st generation:
list_df <- list(tibble(id= UBCoS_m_g1_w2$id, income2=UBCoS_m_g1_w2$income),
                tibble(id= UBCoS_m_g1_w4$id, income4=UBCoS_m_g1_w4$income),

  df <- full_join(x, y, by= "id")
  return(df)
